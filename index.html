// Import the functions you need from the SDKs import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js"; import { getDatabase, ref, push, set, onValue } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js"; // YOUR FIREBASE CONFIGURATION (Paste yours here) const firebaseConfig = { apiKey: "Dolnapha", authDomain: "attendance-app-2026.firebaseapp.com", databaseURL: "https://attendance-app-2026-default-rtbd.firebaseio.com", projectId: "attendance-app-2026", storageBucket: "YOUR_PROJECT_ID.appspot.com", messagingSenderId: "YOUR_ID", appId: "YOUR_APP_ID" }; // Initialize Firebase const app = initializeApp(firebaseConfig); const db = getDatabase(app); // Select elements const nameInput = document.getElementById('userName'); const submitBtn = document.getElementById('submitBtn'); const attendanceTable = document.getElementById('attendanceList'); // Function to save attendance submitBtn.onclick = function() { const name = nameInput.value.trim(); if (name !== "") { const timestamp = new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }); // Push to Firebase (creates a unique ID) const attendanceRef = ref(db, 'attendance'); const newEntryRef = push(attendanceRef); set(newEntryRef, { name: name, time: timestamp }).then(() => { nameInput.value = ""; // Clear input }); } else { alert("Please enter a name"); } }; // Listen for data in real-time onValue(ref(db, 'attendance'), (snapshot) => { const data = snapshot.val(); attendanceTable.innerHTML = ""; // Clear existing UI if (data) { // Loop through data and add rows Object.values(data).forEach(entry => { const row = ` ${entry.name} ${entry.time} `; attendanceTable.innerHTML += row; }); } else { attendanceTable.innerHTML = "No records found."; } });
